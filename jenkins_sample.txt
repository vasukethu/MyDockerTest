pipeline {
    agent any
    stages {
        stage('Git Clone') {
            steps {
                git credentialsId: 'Git-Login', url: 'https://github.com/vasukethu/Dockerpush.git'
            }
            }
 
        stage('Create Tomcat Docker Image') {
            steps {
                sh "docker rm ${docker ps -q}"
                sh "docker build . -t myappnginx"
                sh "docker run -p 82:80 -d --name 'vkapp' myappnginx"
            }
        }
 
    }
}
